const { default: makeWASocket, DisconnectReason, useMultiFileAuthState, } = require("@whiskeysockets/baileys"); const qrcode = require("qrcode-terminal"); const pino = require("pino"); const fs = require("fs"); // ───────────────────────────────────────── // CONFIGURAÇÕES — edite aqui // ───────────────────────────────────────── const CONFIG = { // Horários de expediente (24h) expediente: { manha: { inicio: 9, fim: 12 }, // 09:00 às 12:00 tarde: { inicio: 14, fim: 19 }, // 14:00 às 19:00 diasUteis: [1, 2, 3, 4, 5], // seg=1 ... sab=6, dom=0 }, // Mensagem automática fora do expediente mensagemForaExpediente: `Olá! 👋 Obrigado pelo contato. Nosso horário de atendimento é: 🕘 *Das 9h às 12h* e *das 14h às 19h*, de segunda a sexta. Assim que voltarmos, responderemos sua mensagem o mais breve possível. 🙏 _Mensagem automática_`, // Arquivo com contatos para notificações arquivoClientes: "./clientes.json", }; // ───────────────────────────────────────── // HELPERS // ───────────────────────────────────────── function estaForaDoExpediente() { const agora = new Date(); const hora = agora.getHours(); const minutos = agora.getMinutes(); const horaDecimal = hora + minutos / 60; const diaSemana = agora.getDay(); const diaUtil = CONFIG.expediente.diasUteis.includes(diaSemana); if (!diaUtil) return true; // fim de semana = fora do expediente const noManha = horaDecimal >= CONFIG.expediente.manha.inicio && horaDecimal < CONFIG.expediente.manha.fim; const naTarde = horaDecimal >= CONFIG.expediente.tarde.inicio && horaDecimal < CONFIG.expediente.tarde.fim; return !noManha && !naTarde; // fora se não está nem de manhã nem de tarde } function carregarClientes() { if (!fs.existsSync(CONFIG.arquivoClientes)) { console.log("⚠️ clientes.json não encontrado. Criando arquivo de exemplo..."); const exemplo = [ { nome: "João Silva", telefone: "5511999990001" }, { nome: "Maria Souza", telefone: "5511999990002" }, ]; fs.writeFileSync(CONFIG.arquivoClientes, JSON.stringify(exemplo, null, 2)); } return JSON.parse(fs.readFileSync(CONFIG.arquivoClientes, "utf-8")); } // Controle para não responder o mesmo número 2x no mesmo período const respondidos = new Set(); // Limpa o controle a cada 1h setInterval(() => respondidos.clear(), 60 * 60 * 1000); // ───────────────────────────────────────── // BOT PRINCIPAL // ───────────────────────────────────────── let sock; async function conectar() { const { state, saveCreds } = await useMultiFileAuthState("./sessao"); sock = makeWASocket({ auth: state, logger: pino({ level: "silent" }), printQRInTerminal: false, }); sock.ev.on("connection.update", ({ connection, lastDisconnect, qr }) => { if (qr) { console.log("\n📱 Escaneie o QR Code abaixo com o WhatsApp:\n"); qrcode.generate(qr, { small: true }); } if (connection === "close") { const deveReconectar = lastDisconnect?.error?.output?.statusCode !== DisconnectReason.loggedOut; console.log("🔌 Conexão encerrada. Reconectando:", deveReconectar); if (deveReconectar) conectar(); else console.log("❌ Desconectado permanentemente. Delete a pasta ./sessao e reinicie."); } if (connection === "open") { console.log("✅ Bot conectado ao WhatsApp com sucesso!\n"); console.log("🕘 Expediente: seg-sex, 9h-12h e 14h-19h"); console.log("📋 Para notificar clientes: node notificar.js 'Sua mensagem'\n"); } }); sock.ev.on("creds.update", saveCreds); // ── Resposta automática fora do expediente ── sock.ev.on("messages.upsert", async ({ messages, type }) => { if (type !== "notify") return; for (const msg of messages) { if (msg.key.fromMe) continue; if (msg.key.remoteJid.endsWith("@g.us")) continue; const remetente = msg.key.remoteJid; if (estaForaDoExpediente() && !respondidos.has(remetente)) { try { await sock.sendMessage(remetente, { text: CONFIG.mensagemForaExpediente, }); respondidos.add(remetente); console.log(`📨 Resposta automática enviada para ${remetente}`); } catch (err) { console.error("Erro ao responder:", err.message); } } } }); } module.exports = { conectar, getSock: () => sock, carregarClientes }; conectar();
